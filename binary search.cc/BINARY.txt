
#include<iostream>
using namespace std;

int main()
{
    int n;
    cout<<"Enter size: ";
    cin>>n;

    int arr[n];

    cout<<"Enter sorted array: "<<endl;
    for(int i=0;i<n;i++)
    {
        cin>>arr[i];
    }

    int t;
    cout<<"Enter target: ";
    cin>>t;

    int s=0, e=n-1;

    while(s<=e)
    {
        int mid = (s+e)/2;

        if(arr[mid]==t)
        {
            cout<<"Found"<<mid;
            break;
        }
        else if(arr[mid]<t)
        {
            s = mid + 1;
        }
        else
        {
            e = mid - 1;
        }
    }

    if(s>e)
    {
        cout<<"Not found";
    }

    return 0;
}
